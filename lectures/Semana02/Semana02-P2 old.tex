# Varianza muestral (n=3)

elif unidad_destino == "m":
    result = valor_m
elif unidad_destino == "km":
    result = valor_m / 1000
else:
    result = None

print(f"Resultado: {resultado:.3f} {unidad_destino}")
\end{minted>
