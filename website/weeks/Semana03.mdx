export const FechaLinks = () => {

    const openDate = new Date('2025-03-17T00:00:00');

      setAvailable(true);
    }
  }, []);

  if (available) {
    return (
      <div>

            </blockquote>
          </li>
          <li>
            <a href="/lectures/Semana03-C2.pdf">Semana03-C2.pdf</a>
            <blockquote>
              Este archivo contiene una guía con algunos ejercicios para desarrollar inicialmente, y luego la tarea grupal Semanal.
            </blockquote>
          </li>
        </ul>

        <h3>Material Complementario</h3>
        <p>Configurando un menú contextual "Bash"</p>

        <p>Lo que hace es generar un .EXE en Windows, un .DMG en MAC o el ejecutable que utilice el sistema operativo. Dentro del ejecutable se incluye el propio intérprete de Python, y por esa razón podremos utilizarlo en cualquier ordenador sin necesidad de instalar Python previamente.</p>

conda update anaconda

        </pre>
