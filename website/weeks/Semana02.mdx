import { useState, useEffect } from 'react';

export const FechaLinks = () => {
  const [available, setAvailable] = useState(false);

  useEffect(() => {
    const today = new Date();
    const openDate = new Date('2025-03-10T00:00:00');
    if (today >= openDate) {
      setAvailable(true);
    }
  }, []);

  if (available) {
    return (
      <ul>
        <li>
          <a href="/lectures/Semana02-C1.pdf">Semana02-C1.pdf</a>
          <blockquote>
            Este archivo contiene los <em>slides</em> presentados en la primera clase de la segunda semana. Estos suelen incluir ejercicios al final de cada presentación.
          </blockquote>
        </li>
        <li>
          <a href="/lectures/Semana02-C2.pdf">Semana02-C2.pdf</a>
          <blockquote>
            Este archivo contiene los <em>slides</em> de la segunda clase de esta semana. Además incorpora la actividad de tarea semanal a ser entregada.
          </blockquote>
        </li>
      </ul>
    );
  } else {
    return (
      <strong>El material estará disponible a partir del 10 de marzo de 2025.</strong>
    );
  }
}
